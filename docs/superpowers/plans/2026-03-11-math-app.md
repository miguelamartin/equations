# Math Learning App Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a single self-contained `index.html` math practice app in Spanish for a 12-year-old learning pre-algebra equations, with gamification (points, levels, streaks) and localStorage persistence.

**Architecture:** Single `index.html` with embedded `<style>` and `<script>`. JavaScript is organized into five clearly delimited sections: State, Problem Generator, Answer Checker, Scoring, and UI Controller. No build tools, no dependencies, no server.

**Tech Stack:** HTML5, CSS3, Vanilla JavaScript, localStorage API

```js
// ════════════════════════════════════════════════
// STATE
// ════════════════════════════════════════════════
const STORAGE_KEY = 'math_app_v1';

function defaultState() {
  return { points: 0, level: 1, correctCount: 0, streak: 0, lastDate: null };
}

function loadState() {
  try {
    const raw = localStorage.getItem(STORAGE_KEY);
    return raw ? Object.assign(defaultState(), JSON.parse(raw)) : defaultState();
  } catch { return defaultState(); }
}

function saveState() {
  localStorage.setItem(STORAGE_KEY, JSON.stringify(state));
}

const state = loadState();
updateStreak();

function todayStr() {
  return new Date().toISOString().slice(0, 10); // YYYY-MM-DD
}

function updateStreak() {
  const today = todayStr();
  if (!state.lastDate) { /* first time — do nothing, streak stays 0 */ return; }
  const last = new Date(state.lastDate);
  const now  = new Date(today);
  const diffDays = Math.round((now - last) / 86400000);
  if (diffDays === 1) { state.streak += 1; }
  else if (diffDays > 1) { state.streak = 0; }
  // diffDays === 0 means same day, no change
}

function markPlayedToday() {
  const today = todayStr();
  if (state.lastDate !== today) {
    if (state.lastDate) {
      const last = new Date(state.lastDate);
      const now  = new Date(today);
      const diffDays = Math.round((now - last) / 86400000);
      if (diffDays === 1) state.streak += 1;
      else if (diffDays > 1) state.streak = 0;
    }
    state.lastDate = today;
    saveState();
  }
}
```

// ════════════════════════════════════════════════
// PROBLEM GENERATOR
// ════════════════════════════════════════════════

// Returns a random integer in [min, max] excluding 0 if excludeZero=true
function rnd(min, max, excludeZero = false) {
  let n;
  do { n = Math.floor(Math.random() * (max - min + 1)) + min; }
  while (excludeZero && n === 0);
  return n;
}

// Returns a random non-zero integer in [-max, -1] ∪ [1, max]
function rndNZ(max) { return rnd(1, max) * (Math.random() < 0.5 ? 1 : -1); }

// Format coefficient: 1x -> "x", -1x -> "-x", 2x -> "2x"
function coef(c) {
  if (c === 1) return 'x';
  if (c === -1) return '-x';
  return `${c}x`;
}

// Format sign for constant in expression: e.g. + 3 or - 3
function sign(n) { return n >= 0 ? `+ ${n}` : `- ${Math.abs(n)}`; }

// ── Level 1–2: one-step  ─────────────────────────
// x + b = c  or  ax = c
function genLevel1() {
  if (Math.random() < 0.5) {
    // x + b = c  →  x = c - b
    const b = rnd(-10, 10, true);
    const x = rnd(-10, 10, true);
    const c = x + b;
    const expr = b >= 0 ? `x + ${b}` : `x - ${Math.abs(b)}`;
    return {
      display: `${expr} = ${c}`,
      answer: x,
      explanation: `Despeja x: x = ${c} ${b >= 0 ? '-' : '+'} ${Math.abs(b)} = ${x}`
    };
  } else {
    // ax = c  →  x = c/a
    const a = rndNZ(10);
    const x = rnd(-10, 10, true);
    const c = a * x;
    return {
      display: `${coef(a)} = ${c}`,
      answer: x,
      explanation: `Divide ambos lados entre ${a}: x = ${c} ÷ ${a} = ${x}`
    };
  }
}

// ── Level 3–4: two-step  ─────────────────────────
// ax + b = c
function genLevel2() {
  const a = rndNZ(8);
  const b = rnd(-10, 10, true);
  const x = rnd(-8, 8, true);
  const c = a * x + b;
  const lhs = b >= 0 ? `${coef(a)} + ${b}` : `${coef(a)} - ${Math.abs(b)}`;
  return {
    display: `${lhs} = ${c}`,
    answer: x,
    explanation: `Resta ${b} a ambos lados: ${coef(a)} = ${c - b}. Luego divide entre ${a}: x = ${x}`
  };
}

// ── Level 5–6: negatives & fractions  ────────────
// ax + b = c  with fractions: x/d + b = c
function genLevel3() {
  if (Math.random() < 0.5) {
    // same as level 2 but coefficients can be negative and larger
    const a = rndNZ(10);
    const b = rnd(-15, 15, true);
    const x = rnd(-10, 10, true);
    const c = a * x + b;
    const lhs = b >= 0 ? `${coef(a)} + ${b}` : `${coef(a)} - ${Math.abs(b)}`;
    return {
      display: `${lhs} = ${c}`,
      answer: x,
      explanation: `Resta ${b >= 0 ? b : '(' + b + ')'} a ambos lados: ${coef(a)} = ${c - b}. Divide entre ${a}: x = ${x}`
    };
  } else {
    // x/d + b = c  →  x = (c - b) * d
    const d = rnd(2, 6);
    const b = rnd(-10, 10, true);
    const x = rnd(-5, 5, true) * d; // ensure integer answer
    const c = x / d + b;
    const lhs = b >= 0 ? `x/${d} + ${b}` : `x/${d} - ${Math.abs(b)}`;
    return {
      display: `${lhs} = ${c}`,
      answer: x,
      explanation: `Resta ${b} a ambos lados: x/${d} = ${c - b}. Multiplica por ${d}: x = ${x}`
    };
  }
}
```

// ── Level 7–8: parentheses  ──────────────────────
// a(x + b) = c  or  a(bx + c) = d
function genLevel4() {
  if (Math.random() < 0.5) {
    // a(x + b) = c
    const a = rndNZ(6);
    const b = rnd(-8, 8, true);
    const x = rnd(-8, 8, true);
    const c = a * (x + b);
    return {
      display: `${a}(x ${sign(b)}) = ${c}`,
      answer: x,
      explanation: `Divide entre ${a}: x ${sign(b)} = ${c/a}. Despeja x: x = ${x}`
    };
  } else {
    // a(bx + c) = d
    const a = rnd(2, 5);
    const b = rndNZ(4);
    const c = rnd(-6, 6, true);
    const x = rnd(-5, 5, true);
    const d = a * (b * x + c);
    return {
      display: `${a}(${coef(b)} ${sign(c)}) = ${d}`,
      answer: x,
      explanation: `Divide entre ${a}: ${coef(b)} ${sign(c)} = ${d/a}. Despeja: x = ${x}`
    };
  }
}

// ── Level 9–10: variables on both sides  ─────────
// ax + b = cx + d
function genLevel5() {
  const a = rndNZ(8);
  let c;
  do { c = rndNZ(8); } while (c === a);
  const x = rnd(-8, 8, true);
  const b = rnd(-10, 10, true);
  const d = (a - c) * x + b;
  const lhs = b >= 0 ? `${coef(a)} + ${b}` : `${coef(a)} - ${Math.abs(b)}`;
  const rhs = `${coef(c)} ${sign(d)}`;
  return {
    display: `${lhs} = ${rhs}`,
    answer: x,
    explanation: `Pasa las x a un lado: ${coef(a - c)} = ${d - b}. x = ${x}`
  };
}

// ── Level 11+: combined complexity  ─────────────
// a(bx + c) + dx = e  or  a(x + b) - cx = d(x + e)
function genLevel6() {
  if (Math.random() < 0.5) {
    // a(bx + c) + dx = e
    const a = rnd(2, 4);
    const b = rndNZ(3);
    const c = rnd(-5, 5, true);
    const d = rndNZ(4);
    const x = rnd(-5, 5, true);
    const e = a * (b * x + c) + d * x;
    return {
      display: `${a}(${coef(b)} ${sign(c)}) ${sign(d) === '+ ' + d ? '+ ' + d : '- ' + Math.abs(d)}x = ${e}`,
      answer: x,
      explanation: `Expande: ${a*b}x ${sign(a*c)} ${sign(d)}x = ${e}. Suma: ${a*b+d}x = ${e - a*c}. x = ${x}`
    };
  } else {
    // a(x + b) = c(x + d) — variables both sides with parens
    const a = rnd(2, 6);
    let c;
    do { c = rnd(2, 6); } while (c === a);
    const b = rnd(-5, 5, true);
    const x = rnd(-6, 6, true);
    const d = ((a - c) * x + a * b) / c;
    // Ensure integer d
    if (!Number.isInteger(d)) return genLevel6(); // retry
    return {
      display: `${a}(x ${sign(b)}) = ${c}(x ${sign(d)})`,
      answer: x,
      explanation: `Expande: ${a}x ${sign(a*b)} = ${c}x ${sign(c*d)}. Despeja: ${a-c}x = ${c*d - a*b}. x = ${x}`
    };
  }
}

// ── Dispatcher  ──────────────────────────────────
function generateProblem(level) {
  if (level <= 2)  return genLevel1();
  if (level <= 4)  return genLevel2();
  if (level <= 6)  return genLevel3();
  if (level <= 8)  return genLevel4();
  if (level <= 10) return genLevel5();
  return genLevel6();
}
```

// Parse user input: accepts integers, decimals, simple fractions (e.g. "3/2")
function parseAnswer(raw) {
  const s = raw.trim().replace(',', '.'); // allow comma as decimal separator
  if (/^-?\d+$/.test(s)) return parseInt(s, 10);
  if (/^-?\d+(\.\d+)?$/.test(s)) return parseFloat(s);
  const frac = s.match(/^(-?\d+)\s*\/\s*(\d+)$/);
  if (frac) return parseInt(frac[1], 10) / parseInt(frac[2], 10);
  return NaN;
}

function checkAnswer(userRaw, correctAnswer) {
  const user = parseAnswer(userRaw);
  if (isNaN(user)) return false;
  return Math.abs(user - correctAnswer) < 0.001;
}
```

// ════════════════════════════════════════════════

const POINTS_PER_LEVEL = 10; // base; multiplied by level tier
const CORRECT_FOR_LEVELUP = 10;

function pointsForLevel(level) {
  return level <= 2 ? 5 : level <= 4 ? 8 : level <= 6 ? 10 : level <= 8 ? 13 : level <= 10 ? 15 : 20;
}

// Returns true if a level-up occurred
function registerCorrect() {
  state.points += pointsForLevel(state.level);
  state.correctCount += 1;
  markPlayedToday();
  saveState();
  if (state.correctCount % CORRECT_FOR_LEVELUP === 0) {
    state.level += 1;
    saveState();
    return true; // level-up!
  }
  return false;
}

function registerWrong() {
  markPlayedToday();
  saveState();
}
```

```js
// ════════════════════════════════════════════════
// UI CONTROLLER
// ════════════════════════════════════════════════

let currentProblem = null;

// DOM refs
const elProblem   = document.getElementById('problem');
const elAnswer    = document.getElementById('answer');
const elSubmit    = document.getElementById('submit');
const elFeedback  = document.getElementById('feedback');
const elNext      = document.getElementById('next');
const elLevelup   = document.getElementById('levelup');
const elLuLevel   = document.getElementById('lu-level');
const elLuClose   = document.getElementById('lu-close');
const elHdrPoints = document.getElementById('hdr-points');
const elHdrStreak = document.getElementById('hdr-streak');
const elLevelNum  = document.getElementById('level-num');

function refreshHeader() {
  elHdrPoints.textContent = state.points;
  elHdrStreak.textContent = state.streak;
  elLevelNum.textContent  = state.level;
}

function showProblem() {
  currentProblem = generateProblem(state.level);
  elProblem.textContent = currentProblem.display;
  elAnswer.value = '';
  elAnswer.disabled = false;
  elFeedback.className = '';
  elFeedback.style.display = 'none';
  elFeedback.innerHTML = '';
  elNext.style.display = 'none';
  elSubmit.style.display = '';
  refreshHeader();
  elAnswer.focus();
}

function onSubmit() {
  const raw = elAnswer.value;
  if (!raw.trim()) return;

  elAnswer.disabled = true;
  elSubmit.style.display = 'none';

  const correct = checkAnswer(raw, currentProblem.answer);

  if (correct) {
    const leveled = registerCorrect();
    refreshHeader();
    elFeedback.className = 'correct';
    elFeedback.innerHTML = `<strong>Correcto!</strong>${currentProblem.explanation}`;
    elFeedback.style.display = 'block';
    elNext.style.display = 'block';
    if (leveled) {
      setTimeout(() => {
        elLuLevel.textContent = state.level;
        elLevelup.classList.add('show');
      }, 600);
    }
  } else {
    registerWrong();
    elFeedback.className = 'wrong';
    elFeedback.innerHTML = `<strong>Incorrecto.</strong>La respuesta correcta es x = ${currentProblem.answer}.<br>${currentProblem.explanation}`;
    elFeedback.style.display = 'block';
    elNext.style.display = 'block';
  }
}

// Event listeners
elSubmit.addEventListener('click', onSubmit);
elAnswer.addEventListener('keydown', e => { if (e.key === 'Enter') onSubmit(); });
elNext.addEventListener('click', showProblem);
elLuClose.addEventListener('click', () => {
  elLevelup.classList.remove('show');
  showProblem();
});

// Boot
refreshHeader();
showProblem();
```
